-- 1
select p.ProductName, p.UnitPrice from products as p
order by p.UnitPrice desc
limit 3

-- 2

select p.ProductName, s.CompanyName from products as p
join suppliers as s on s.supplierId = p.SupplierID
order by p.ProductName
limit 5

-- 3 
select em.FirstName, em.LastName from employees as e
join employees as em on em.EmployeeID = e.ReportsTo
group by e.ReportsTo
having count(e.EmployeeID) > 2

-- 4 

select p.ProductName, p.UnitPrice, c.CategoryName 
from products as p
join categories as c on c.CategoryID = p.CategoryID
where p.UnitPrice = (
                    select max(pr.UnitPrice) from products as pr
                     where pr.UnitPrice = p.UnitPrice 
							and pr.CategoryID = p.CategoryID
                    )
order by p.UnitPrice
